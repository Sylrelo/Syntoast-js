Syntoast Demo

Success message 

toast.show({
	type: 'success',
	title: 'Success !',
	text: 'Eveything went well :)'
})

Error message 

toast.show({
	type: 'error',
	title: 'Error !',
	text: 'Everything is in flame !'
})

warning message 

toast.show({
	type: 'warning',
	title: 'Be careful !',
	text: 'Someone is watching you...'
})

Info message 

toast.show({
	type: 'info',
	title: 'Uh ?',

})

	type: 'Here she come',
	title: 'Eleven',
